
# 上下采样（两种方法）
## 直接变换形状（如patch）。
### patch
```python
def img2patch(x: torch.Tensor, nh: int, nw: int):
    """
    只切空间H、W，时间T不动
    :param x: 输入 [B, T, C, H, W]
    :param nh: 高度分成 nh 块
    :param nw: 宽度分成 nw 块
    :return: patches [B*nh*nw, T, C, ph, pw]
    """


    B, T, C, H, W = x.shape
    x = x[:,:,:,:H,:W]
    ph = H // nh
    pw = W // nw

    # 1.拆分空间维度，B,T,C保持不变
    x = x.reshape(B, T, C, nh, ph, nw, pw)
    # 2.把分块维度换到前面
    x = x.permute(0, 3, 5, 1, 2, 4, 6)
    # 3.把所有小块压进batch维
    patches = x.reshape(B * nh * nw, T, C, ph, pw)
    # print("patches.shape:",patches.shape)
    return patches


def patch2img(patches: torch.Tensor, B: int, nh: int, nw: int):
    """
    把小块拼回原图
    :param patches: [B*nh*nw, T, C, ph, pw]
    :param B: 原始批次大小
    :param nh,nw: 分块数量
    :return: 原图 [B, T, C, H, W]
    """
    total_patch = nh * nw
    _, T, C, ph, pw = patches.shape

    # 恢复网格结构
    x = patches.reshape(B, nh, nw, T, C, ph, pw)
    # 换回原始维度顺序
    x = x.permute(0, 3, 4, 1, 5, 2, 6)
    # 合并高度、宽度
    B, T, C, _, _, _, _ = x.shape
    H = nh * ph
    W = nw * pw
    img = x.reshape(B, T, C, H, W)
    return img
```

## 用3d卷积变换（如UNet）
```python
class SpatioDownSample(nn.Module):
    def __init__(self, in_channels, out_channels):
        super().__init__()
        self.down_conv = nn.Conv3d(
            in_channels, out_channels,
            kernel_size=(3,3,3),
            stride=(1,2,2),
            padding=(1,1,1),
            bias=True
        )
        self.norm = nn.BatchNorm3d(out_channels)
        self.relu = nn.ReLU(inplace=True)

    def forward(self, x):
        return self.relu(self.norm(self.down_conv(x)))

class SpatioUpSample(nn.Module):
    def __init__(self, in_channels, out_channels):
        super().__init__()
        self.up_conv = nn.ConvTranspose3d(
            in_channels, out_channels,
            kernel_size=(3,3,3),
            stride=(1,2,2),
            padding=(1,1,1),
            output_padding=(0,1,1),
            bias=True
        )
        self.norm = nn.BatchNorm3d(out_channels)
        self.relu = nn.ReLU(inplace=True)

    def forward(self, x):
        return self.relu(self.norm(self.up_conv(x)))
```
